33      @Test
34      public void addAndRemoveWiskyAndBook() {
35          Storage storage = new Storage();
36          storage.add(createWiskyBottle());
37          assertEquals(5, storage.calculateWeight().intValue());
38          storage.add(createBook());
39          assertEquals(9, storage.calculateWeight().intValue());
40  
41          List<Item> items = storage.search("wisky");
42          assertNotNull(items);
43          assertEquals(items.size(), 1);
44          Item wisky = items.get(0);
45          assertNotNull(wisky);
46          assertEquals(5, wisky.getWeight().intValue());
47  
48          items = storage.search("book");
49          assertNotNull(items);
50          assertEquals(items.size(), 1);
51          Item book = items.get(0);
52          assertNotNull(book);
53          assertEquals(4, book.getWeight().intValue());
54  
55          assertEquals(9, storage.calculateWeight().intValue());
56  
57          storage.remove(book);
58          assertEquals(5, storage.calculateWeight().intValue());
59          storage.remove(book);
60          assertEquals(5, storage.calculateWeight().intValue());
61          storage.remove(wisky);
62          assertEquals(0, storage.calculateWeight().intValue());
63      }
