61      private void load() {
62          String fileName = "json/npcs.json";
63          JsonParser parser = new JsonParser();
64          File f = new File(fileName);
65          try {
66              Reader reader = new FileReader(fileName);
67              JsonObject json = parser.parse(reader).getAsJsonObject();
68              json = json.get("npcs").getAsJsonObject();
69              Set<Map.Entry<String, JsonElement>> entries = json.entrySet();
70              for (Map.Entry<String, JsonElement> entry : entries) {
71                  NPC npc = new NPC(entry.getKey());
72                  JsonObject details = entry.getValue().getAsJsonObject();
73                  if (details.get("conversations") != null) {
74                      JsonArray conversation = details.get("conversations").getAsJsonArray();
75                      addConversation(npc, conversation);
76                  }
77              }
78          } catch (IOException ex) {
79              ex.printStackTrace();
80          }
81      }
82  
83      private void addConversation(NPC npc, JsonArray conversation) {
84          List<Line> start = new ArrayList<>();
85          int i = 0;
86          for (JsonElement entry : conversation) {
87              JsonObject details = entry.getAsJsonObject();
88              if ("".equals(details.get("player").getAsString())) {
89                  start.add(getLine(i++, conversation));
90              }
91          }
92          lines.put(npc, start);
93      }
94  
95      private Line getLine(int index, JsonArray conversation) {
96          JsonObject line = conversation.get(index).getAsJsonObject();
97          List<Line> responses = new ArrayList<>();
98          if (line.get("response") != null) {
99              for (JsonElement i : line.get("response").getAsJsonArray()) {
100                 responses.add(getLine(i.getAsInt(), conversation));
101             }
102         }
103         String playerPrompt = line.get("player").getAsString();
104         String text = line.get("text").getAsString();
105         String[] con = line.get("condition").getAsString().split("=");
106         ConditionType condition = CONDITION_TYPE_MAP.get(con[0]);
107         String conditionParameter = (con.length == 1) ? "" : con[1];
108         ActionType action = ACTION_TYPE_MAP.get(line.get("action").getAsString());
109         return new Line(index, playerPrompt, text, condition, conditionParameter, responses, action);
110     }

112     public void startConversation(NPC npc, Player player) throws DeathException {
113         List<Line> conversation = null;
114         //Workaround as <code>lines.get(npc)</code> is not working.
115         Iterator it = lines.entrySet().iterator();
116         while (it.hasNext()) {
117             @SuppressWarnings("unchecked")
118             Map.Entry<NPC, List<Line>> entry = (Map.Entry<NPC, List<Line>>) it.next();
119             if (entry.getKey().equals(npc)) {
120                 conversation = entry.getValue();
121             }
122             it.remove();
123         }
124         if (conversation != null) {
125             Line start = null;
126             for (Line l : conversation) {
127                 if (ConversationManager.matchesConditions(npc, player, l)) {
128                     start = l;
129                     break;
130                 }
131             }
132             if (start != null) {
133                 QueueProvider.offer(start.getText());
134                 Line response = start.display(npc, player);
135                 triggerAction(start, npc, player);
136                while (response != null) {
137                    QueueProvider.offer(response.getText());
138                    triggerAction(response, npc, player);
139                    Line temp_response = response.display(npc, player);
140                    response = temp_response;
141                }
142             }
143         }
144     }

146     private void triggerAction(Line line, NPC npc, Player player) throws DeathException {
147         switch (line.getAction()) {
148             case ATTACK:
149                 QueueProvider.offer("\n" + npc.getName() + " is now attacking you!\n");
150                 player.attack(npc.getName());
151                 break;
152             case TRADE:
153                 Trading t = new Trading(npc, player);
154                 t.trade(true, true);
155                 break;
156         }     
157     }

159     public static boolean matchesConditions(NPC npc, Player player, Line line) {
160         switch(line.getCondition()) {
161             case ALLY:
162                 return npc.getAllies().contains(player.getCurrentCharacterType());
163             case ENEMY:
164                 return npc.getEnemies().contains(player.getCurrentCharacterType());
165             case LEVEL:
166                 int requiredLevel = Integer.parseInt(line.getConditionParameter());
167                 return player.getLevel() >= requiredLevel;
168             case ITEM:
169                 ItemRepository itemRepo = GameBeans.getItemRepository();
170                 Item requiredItem = itemRepo.getItem(line.getConditionParameter());
171                 return player.hasItem(requiredItem);
172             case CHAR_TYPE:
173                 String charType = line.getConditionParameter();
174                 return charType.equals(player.getCurrentCharacterType());
175             default: // No condition
176                 return true;
177         }
178     }
