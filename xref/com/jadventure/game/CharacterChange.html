20      public void trigger(Player player, String triggerType, String keyword) {
21          JsonParser parser = new JsonParser();
22          String fileName = "json/character_transitions.json";
23          try {
24              Reader reader = new FileReader(fileName);
25              JsonObject json = parser.parse(reader).getAsJsonObject();
26  
27              String currentCharacter = player.getCurrentCharacterType();
28  
29              JsonObject currentCharacterTransitions;
30              JsonObject events;
31              JsonObject characterEffects = new JsonObject();
32              boolean goAhead = false;
33              if (json.has(currentCharacter)) {
34                  currentCharacterTransitions = json.get(currentCharacter).getAsJsonObject();
35                  if (currentCharacterTransitions.has(triggerType)) {
36                      events = currentCharacterTransitions.get(triggerType).getAsJsonObject();
37                      if (events.has(keyword)) {
38                          characterEffects = events.get(keyword).getAsJsonObject();
39                          goAhead = true;
40                      } else {
41                          //QueueProvider.offer("Warning: The effects for the '" + triggerType + "' event and the '" + currentCharacter + "' character was not found");
42                      }
43                  } else {
44                      //QueueProvider.offer("Warning: The event '" + triggerType + "' for the '" + currentCharacter + "' character was not found");
45                  }
46              } else {
47                  //QueueProvider.offer("Warning: The character '" + currentCharacter + "' was not found");
48              }
49  
50              if (goAhead == true) {
51                  for (Map.Entry<String, JsonElement> entry : characterEffects.entrySet()) {
52                      String characterName = entry.getKey();
53                      int characterLevelEffect = entry.getValue().getAsInt();
54                      int characterLevel = player.getCharacterLevel(characterName); 
55                      int newCharacterLevel = characterLevel + characterLevelEffect;
56                      player.setCharacterLevel(characterName, newCharacterLevel);
57                      checkForCharacterChange(player);
58                  }
59              }
60  
61          } catch (FileNotFoundException ex) {
62              ex.printStackTrace();
63          }
64      }

66      public void checkForCharacterChange(Player player) {
67          HashMap<String, Integer> characterLevels = player.getCharacterLevels();
68          String currentCharacter = player.getCurrentCharacterType();
69          int highestCharacterLevel = player.getCharacterLevel(currentCharacter);
70          String highestCharacter = currentCharacter;
71          Iterator it = characterLevels.entrySet().iterator();
72          while (it.hasNext()) {
73              Map.Entry pairs = (Map.Entry)it.next();
74              int value = (int)pairs.getValue();
75              if (value > highestCharacterLevel) {
76                  highestCharacterLevel = value;
77                  highestCharacter = (String)pairs.getKey();
78              }
79          }
80          if (!highestCharacter.equals(currentCharacter)) {
81              player.setCurrentCharacterType(highestCharacter);
82              QueueProvider.offer("You're character type is now changed! You are now a " + highestCharacter + "!");
83          }
84          it = characterLevels.entrySet().iterator();
85          while (it.hasNext()) {
86              Map.Entry pairs = (Map.Entry)it.next();
87              player.setCharacterLevel((String)pairs.getKey(), (int)pairs.getValue());
88          }
89      }
