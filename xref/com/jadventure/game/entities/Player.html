105     public static Player load(String name) {
106         player = new Player();
107         JsonParser parser = new JsonParser();
108         String fileName = getProfileFileName(name);
109         try {
110             Reader reader = new FileReader(fileName);
111             JsonObject json = parser.parse(reader).getAsJsonObject();
112             player.setName(json.get("name").getAsString());
113             player.setHealthMax(json.get("healthMax").getAsInt());
114             player.setHealth(json.get("health").getAsInt());
115             player.setArmour(json.get("armour").getAsInt());
116             player.setDamage(json.get("damage").getAsInt());
117             player.setLevel(json.get("level").getAsInt());
118             player.setXP(json.get("xp").getAsInt());
119             player.setStrength(json.get("strength").getAsInt());
120             player.setIntelligence(json.get("intelligence").getAsInt());
121             player.setDexterity(json.get("dexterity").getAsInt());
122             player.setLuck(json.get("luck").getAsInt());
123             player.setStealth(json.get("stealth").getAsInt());
124             player.setCurrentCharacterType(json.get("type").getAsString());
125             HashMap<String, Integer> charLevels = new Gson().fromJson(json.get("types"), new TypeToken<HashMap<String, Integer>>(){}.getType());
126             player.setCharacterLevels(charLevels);
127             player.equipItem(EquipmentLocation.RIGHT_HAND, itemRepo.getItem((json.get("weapon").getAsString())));
128             if (json.has("items")) {
129                 HashMap<String, Integer> items = new Gson().fromJson(json.get("items"), new TypeToken<HashMap<String, Integer>>(){}.getType());
130                 ArrayList<ItemStack> itemList = new ArrayList<ItemStack>();
131                 for (Map.Entry<String, Integer> entry : items.entrySet()) {
132                     String itemID = entry.getKey();
133                     int amount = entry.getValue();
134                     Item item = itemRepo.getItem(itemID);
135                     ItemStack itemStack = new ItemStack(amount, item);
136                     itemList.add(itemStack);
137                 }
138                 float maxWeight = (float)Math.sqrt(player.getStrength()*300);
139                 player.setStorage(new Storage(maxWeight, itemList));
140             }
141             Coordinate coordinate = new Coordinate(json.get("location").getAsString());
142             LocationManager.getInstance(player.getName());
143             player.setLocation(LocationManager.getLocation(coordinate));
144             reader.close();
145             setUpCharacterLevels();
146         } catch (FileNotFoundException ex) {
147             QueueProvider.offer( "Unable to open file '" + fileName + "'.");
148         } catch (IOException ex) {
149             ex.printStackTrace();
150         }
151 
152         return player;
153     }

158     public static Player getInstance(String playerClass){
159         player = new Player();
160         JsonParser parser = new JsonParser();
161         String fileName = "json/npcs.json";
162         try {
163             Reader reader = new FileReader(fileName);
164             JsonObject npcs = parser.parse(reader).getAsJsonObject().get("npcs").getAsJsonObject();
165             JsonObject json = new JsonObject();
166             for (Map.Entry<String, JsonElement> entry : npcs.entrySet()) {
167                 if (entry.getKey().equals(playerClass)) {
168                     json = entry.getValue().getAsJsonObject();
169                 }
170             }
171 
172             player.setName(json.get("name").getAsString());
173             player.setHealthMax(json.get("healthMax").getAsInt());
174             player.setHealth(json.get("health").getAsInt());
175             player.setArmour(json.get("armour").getAsInt());
176             player.setDamage(json.get("damage").getAsInt());
177             player.setLevel(json.get("level").getAsInt());
178             player.setXP(json.get("xp").getAsInt());
179             player.setStrength(json.get("strength").getAsInt());
180             player.setIntelligence(json.get("intelligence").getAsInt());
181             player.setDexterity(json.get("dexterity").getAsInt());
182             setUpVariables(player);
183             JsonArray items = json.get("items").getAsJsonArray();
184             for (JsonElement item : items) {
185                 player.addItemToStorage(itemRepo.getItem(item.getAsString()));
186             }
187             Random rand = new Random();
188             int luck = rand.nextInt(3) + 1;
189             player.setLuck(luck);
190             player.setStealth(json.get("stealth").getAsInt());
191             player.setIntro(json.get("intro").getAsString());
192             if (player.getName().equals("Recruit")) {
193                 player.type = "Recruit";
194             } else if (player.getName().equals("Sewer Rat")) {
195                 player.type = "Sewer Rat";
196             } else {
197                 QueueProvider.offer("Not a valid class");
198             }
199             reader.close();
200             setUpCharacterLevels();
201         } catch (FileNotFoundException ex) {
202             QueueProvider.offer( "Unable to open file '" + fileName + "'.");
203         } catch (IOException ex) {
204             ex.printStackTrace();
205         }
206 
207         return player;
208     }

249     public void save() {
250         Gson gson = new Gson();
251         JsonObject jsonObject = new JsonObject();
252         jsonObject.addProperty("name", getName());
253         jsonObject.addProperty("healthMax", getHealthMax());
254         jsonObject.addProperty("health", getHealthMax());
255         jsonObject.addProperty("armour", getArmour());
256         jsonObject.addProperty("damage", getDamage());
257         jsonObject.addProperty("level", getLevel());
258         jsonObject.addProperty("xp", getXP());
259         jsonObject.addProperty("strength", getStrength());
260         jsonObject.addProperty("intelligence", getIntelligence());
261         jsonObject.addProperty("dexterity", getDexterity());
262         jsonObject.addProperty("luck", getLuck());
263         jsonObject.addProperty("stealth", getStealth());
264         jsonObject.addProperty("weapon", getWeapon());
265         jsonObject.addProperty("type", getCurrentCharacterType());
266         HashMap<String, Integer> items = new HashMap<String, Integer>();
267         JsonArray itemList = new JsonArray();
268         for (ItemStack item : getStorage().getItems()) {
269             items.put(item.getItem().getId(), item.getAmount());
270             JsonPrimitive itemJson = new JsonPrimitive(item.getItem().getId());
271             itemList.add(itemJson);
272         }
273         JsonElement itemsJsonObj = gson.toJsonTree(items);
274         jsonObject.add("items", itemsJsonObj);
275         JsonElement typesJsonObj = gson.toJsonTree(getCharacterLevels());
276         jsonObject.add("types", typesJsonObj);
277         Coordinate coordinate = getLocation().getCoordinate();
278         String coordinateLocation = coordinate.x+","+coordinate.y+","+coordinate.z;
279         jsonObject.addProperty("location", coordinateLocation);
280 
281         String fileName = getProfileFileName(getName());
282         new File(fileName).getParentFile().mkdirs();
283         try {
284             Writer writer = new FileWriter(fileName);
285             gson.toJson(jsonObject, writer);
286             writer.close();
287             LocationManager.writeLocations(getName());
288             QueueProvider.offer("\nYour game data was saved.");
289         } catch (IOException ex) {
290             QueueProvider.offer("\nUnable to save to file '" + fileName + "'.");
291         }
292     }

337     public void dropItem(String itemName) {
338         List<Item> itemMap = searchItem(itemName, getStorage());
339         if (itemMap.isEmpty()) {
340             itemMap = searchEquipment(itemName, getEquipment());
341         }
342         if (!itemMap.isEmpty()) {
343             Item item = itemMap.get(0);
344             Item itemToDrop = itemRepo.getItem(item.getId());
345             Item weapon = itemRepo.getItem(getWeapon());
346             String wName = weapon.getName();
347 
348             if (itemName.equals(wName)) {
349                 dequipItem(wName);
350             }
351             removeItemFromStorage(itemToDrop);
352             location.addPublicItem(itemToDrop.getId());
353             QueueProvider.offer(item.getName() + " dropped");
354         }
355     }

357     public void equipItem(String itemName) {
358         List<Item> itemMap = searchItem(itemName, getStorage());
359         if (!itemMap.isEmpty()) {
360             Item item = itemMap.get(0);
361             if (getLevel() >= item.getLevel()) {
362                 Map<String, String> change = equipItem(item.getPosition(), item);
363                 QueueProvider.offer(item.getName()+ " equipped");
364                 printStatChange(change);
365             } else {
366                 QueueProvider.offer("You do not have the required level to use this item");
367             }
368         } else {
369             QueueProvider.offer("You do not have that item");
370         }
371     }

383     private void printStatChange(Map<String, String> stats) {
384          Set<Entry<String, String>> set = stats.entrySet();
385          Iterator<Entry<String, String>> i = set.iterator();
386          while (i.hasNext()) {
387               Entry<String, String> me = i.next();
388               double value = Double.parseDouble((String) me.getValue());
389               switch ((String) me.getKey()) {
390                   case "damage": {
391                           if (value >= 0.0) {
392                               QueueProvider.offer(me.getKey() + ": " + this.getDamage() + " (+" + me.getValue() + ")");
393                           } else {
394                               QueueProvider.offer(me.getKey() + ": " + this.getDamage() + " (" + me.getValue() + ")");
395                           }
396                           break;
397                     }
398                     case "health": {
399                           if (value >= 0) {
400                               QueueProvider.offer(me.getKey() + ": " + this.getHealth() + " (+" + me.getValue() + ")");
401                           } else {
402                               QueueProvider.offer(me.getKey() + ": " + this.getHealth() + " (" + me.getValue() + ")");
403                           }
404                           break;
405                     }
406                     case "armour": {
407                           if (value >= 0) {
408                               QueueProvider.offer(me.getKey() + ": " + this.getArmour() + " (+" + me.getValue() + ")");
409                           } else {
410                               QueueProvider.offer(me.getKey() + ": " + this.getArmour() + " (" + me.getValue() + ")");
411                           }
412                           break;
413                     }
414                     case "maxHealth": {
415                           if (value  >= 0) {
416                               QueueProvider.offer(me.getKey() + ": " + this.getHealthMax() + " (+" + me.getValue() + ")");
417                           } else {
418                               QueueProvider.offer(me.getKey() + ": " + this.getHealthMax() + " (" + me.getValue() + ")");
419                           }
420                           break;
421                     }
422               }
423          }
424     }

451     public void attack(String opponentName) throws DeathException {
452         Monster monsterOpponent = null;
453         NPC npcOpponent = null;
454         List<Monster> monsters = getLocation().getMonsters();
455         List<NPC> npcs = getLocation().getNPCs();
456         for (int i = 0; i < monsters.size(); i++) {
457              if (monsters.get(i).monsterType.equalsIgnoreCase(opponentName)) {
458                  monsterOpponent = monsters.get(i);
459              }
460         }
461         for (int i=0; i < npcs.size(); i++) {
462             if (npcs.get(i).getName().equalsIgnoreCase(opponentName)) {
463                 npcOpponent = npcs.get(i);
464             }
465         }
466         if (monsterOpponent != null) {
467             monsterOpponent.setName(monsterOpponent.monsterType);
468             new BattleMenu(monsterOpponent, this);
469         } else if (npcOpponent != null) {
470             new BattleMenu(npcOpponent, this);
471         } else {
472              QueueProvider.offer("Opponent not found");
473         }
474     }
