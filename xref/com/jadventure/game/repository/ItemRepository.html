54      // Load all items, from the given file
55      protected void load(JsonReader reader) {
56          JsonObject jsonItems = new JsonObject();
57          try {
58              JsonParser parser = new JsonParser();
59              JsonObject json = parser.parse(reader).getAsJsonObject();
60              jsonItems = json.get("items").getAsJsonObject();
61              reader.close();
62          } catch (FileNotFoundException ex) {
63              ex.printStackTrace();
64          } catch (IOException ex) {
65              ex.printStackTrace();
66          }
67          
68          for (Map.Entry<String, JsonElement> entry : jsonItems.entrySet()) {
69              String id = entry.getKey();
70              JsonObject itemData = entry.getValue().getAsJsonObject();
71              String type = itemData.get("type").getAsString();
72              String name = itemData.get("name").getAsString();
73              String description = itemData.get("description").getAsString();
74              EquipmentLocation position = EQUIPMENT_POSITION_MAP.get(itemData.get("position").getAsString());
75              int level = itemData.get("level").getAsInt();
76              JsonObject sProps = itemData.get("properties").getAsJsonObject();
77              Map<String, Integer> properties = new HashMap<>();
78              for (Map.Entry<String, JsonElement> entry2 : sProps.entrySet()) {
79                  Integer propValue = entry2.getValue().getAsInt();
80                  properties.put(entry2.getKey(), propValue);
81              }
82  
83              addItem(new Item(id, type, name, description, position, level, properties));
84              
85          }
86      }

94      public static ItemRepository createRepo() {
95          if (itemRepository == null) {
96              File file = new File(new File(System.getProperty("user.dir")), "json");
97  
98              File dataFile = new File(new File(file, "original_data"), "items.json");
99              if (! dataFile.exists()) {
100             	throw new RuntimeException("File '" + dataFile + "' does not exist.");
101             }
102 
103             itemRepository = new ItemRepository();
104             itemRepository.load(dataFile);
105         }
106         return itemRepository;
107     }
